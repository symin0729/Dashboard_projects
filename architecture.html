function fmtUSD(n){
  if(n===null||n===undefined) return '—';
  return '$' + Number(n).toLocaleString('en-US',{maximumFractionDigits:0});
}
function fmtNum(n){
  if(n===null||n===undefined) return '—';
  return Number(n).toLocaleString('en-US');
}

그래서 data.js 파일은 왜 없나요?

다른 프로젝트에서는 데이터를 data.js 같은 별도 파일로 분리해서 <script src="data.js">로 불러오는 방식을 많이 씁니다. 이 프로젝트는 그 대신 위 다이어그램의 파란 박스(const DATA = {…})처럼 index.html 안에 데이터를 그대로 박아 넣는 방식을 택했습니다 — 파일 하나만 있으면 어디서든 그대로 열리게 하려는 의도적인 선택입니다.
